{{- if and (eq hugo.Environment "production") (not hugo.IsServer) -}} {{- $placeId := site.Params.google.place_id -}} {{- $apiKey := site.Params.google.api_key -}} {{- if and $placeId $apiKey -}} {{- $url := printf "https://maps.googleapis.com/maps/api/place/details/json?place_id=%s&fields=rating&key=%s" (urlquery $placeId) (urlquery $apiKey) -}} {{- with resources.GetRemote $url -}} {{- $json := .Content | transform.Unmarshal -}} {{- if and (eq $json.status "OK") $json.result.rating -}} {{- with $json.result -}} {{- $rating := .rating -}} {{- $full := int (math.Round $rating) -}} {{- $empty := sub 5 $full -}} {{- $stars := printf "%s%s" (strings.Repeat $full "★") (strings.Repeat $empty "☆") -}}
{{ $stars }} Rated {{ $rating }} out of 5 stars — based on Google Reviews
{{- end -}} {{- end -}} {{- end -}} {{- end -}} {{- end -}}
